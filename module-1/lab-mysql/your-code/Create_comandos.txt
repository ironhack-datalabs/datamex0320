#Crear tabla cars en workbench

CREATE TABLE `lab_mysql`.`cars` (
  `id` INT NOT NULL,
  `VIN` VARCHAR(45) NOT NULL,
  `Manufacturer` VARCHAR(45) NOT NULL,
  `Model` VARCHAR(45) NOT NULL,
  `Year` VARCHAR(45) NOT NULL,
  `Color` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`Manufacturer`),
  UNIQUE INDEX `VIN id_UNIQUE` (`VIN` ASC) VISIBLE)
COMMENT = 'Local insurances';

#Crear tabla customers

CREATE TABLE `lab_mysql`.`customer` (
  `id` INT NOT NULL,
  `insuranceid` VARCHAR(45) NOT NULL,
  `brand` VARCHAR(45) NOT NULL,
  `phone` VARCHAR(45) NOT NULL,
  `email` VARCHAR(45) NOT NULL,
  `Addres` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`insuranceid`),
  UNIQUE INDEX `aseguradora id_UNIQUE` (`insuranceid` ASC) VISIBLE)
